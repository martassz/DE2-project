235/* instruction register bit positions, see HD44780U data sheet */
236#define LCD_CLR 0 /* DB0: clear display */
237#define LCD_HOME 1 /* DB1: return to home position */
238#define LCD_ENTRY_MODE 2 /* DB2: set entry mode */
239#define LCD_ENTRY_INC 1 /* DB1: 1=increment, 0=decrement */
240#define LCD_ENTRY_SHIFT 0 /* DB2: 1=display shift on */
241#define LCD_ON 3 /* DB3: turn lcd/cursor on */
242#define LCD_ON_DISPLAY 2 /* DB2: turn display on */
243#define LCD_ON_CURSOR 1 /* DB1: turn cursor on */
244#define LCD_ON_BLINK 0 /* DB0: blinking cursor ? */
245#define LCD_MOVE 4 /* DB4: move cursor/display */
246#define LCD_MOVE_DISP 3 /* DB3: move display (0-> cursor) ? */
247#define LCD_MOVE_RIGHT 2 /* DB2: move right (0-> left) ? */
248#define LCD_FUNCTION 5 /* DB5: function set */
249#define LCD_FUNCTION_8BIT 4 /* DB4: set 8BIT mode (0->4BIT mode) */
250#define LCD_FUNCTION_2LINES 3 /* DB3: two lines (0->one line) */
251#define LCD_FUNCTION_10DOTS 2 /* DB2: 5x10 font (0->5x7 font) */
252#define LCD_CGRAM 6 /* DB6: set CG RAM address */
253#define LCD_DDRAM 7 /* DB7: set DD RAM address */
254#define LCD_BUSY 7 /* DB7: LCD is busy */
255
256/* set entry mode: display shift on/off, dec/inc cursor move direction */
257#define LCD_ENTRY_DEC 0x04 /* display shift off, dec cursor move dir */
258#define LCD_ENTRY_DEC_SHIFT 0x05 /* display shift on, dec cursor move dir */
259#define LCD_ENTRY_INC_ 0x06 /* display shift off, inc cursor move dir */
260#define LCD_ENTRY_INC_SHIFT 0x07 /* display shift on, inc cursor move dir */
261
262/* display on/off, cursor on/off, blinking char at cursor position */
263#define LCD_DISP_OFF 0x08 /* display off */
264#define LCD_DISP_ON 0x0C /* display on, cursor off */
265#define LCD_DISP_ON_BLINK 0x0D /* display on, cursor off, blink char */
266#define LCD_DISP_ON_CURSOR 0x0E /* display on, cursor on */
267#define LCD_DISP_ON_CURSOR_BLINK 0x0F /* display on, cursor on, blink char */
268
269/* move cursor/shift display */
270#define LCD_MOVE_CURSOR_LEFT 0x10 /* move cursor left (decrement) */
271#define LCD_MOVE_CURSOR_RIGHT 0x14 /* move cursor right (increment) */
272#define LCD_MOVE_DISP_LEFT 0x18 /* shift display left */
273#define LCD_MOVE_DISP_RIGHT 0x1C /* shift display right */
